    [root@iZj6cbyftexodnjrn2gt3aZ redis]# cat redis-svc.yaml
---
apiVersion: v1
kind: Service
metadata:
  name: kube-redis-exporter
  namespace: kube-system
  labels:
    app: kube-redis-exporter # service的标签，后面会被serviceMonitor用到
spec:
  type: ClusterIP
  clusterIP: None
  ports:
  - name: redis-metrics # 这个名称一会serviceMonitor会用到

    targetPort: 9121
    protocol: TCP

    [root@iZj6cbyftexodnjrn2gt3aZ manifests]# cat kubernetes-serviceMonitorRedis.yaml
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  labels:
    app: redis-exporter
  name: redis-exporter-scraping
  namespace: kube-system
spec:
  endpoints:
  - interval: 30s
    path: /metrics # 获取数据的接口
    port: redis-metrics # 和之前service里定义的名字要一致
  jobLabel: redis-exporter
  namespaceSelector:
    matchNames:
    - kube-system
  selector:
    matchLabels:
      app: kube-redis-exporter # 通过这个label匹配service，以达到动态发现的目的。

func GetTime() string {
    date := time.Now().Format("2006-01-02 15:04:05")
    local, _ := time.LoadLocation("Local")
    result, _ := time.ParseInLocation("2006-01-02 15:04:05", date, local)
    sd, _ := time.ParseDuration("0h")
    StartTime := result.In(local).Add(sd).Format("2006-01-02 15:04:05")
    return StartTime
}
