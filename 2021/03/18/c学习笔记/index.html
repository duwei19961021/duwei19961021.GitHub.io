 * 数组名是一个常量，不能进行自增。
 * 所以这里的写法是错误的。
 */

void strcpy_best2(char *src, char*dest)
{
    printf("&src in = %p\n", &src);
    printf("src = %p,line = %d\n",src,__LINE__);
    *(*&src+1) = 'A';
    while ((*dest++ = *src++)){
        ;
    }
    /*
     * 在此函数被调用时，数组的地址会拷贝给形参变量，src、dest是变量，所以能够进行自增
     * */
}

int main()
{
    char dest[10];
    char src[] = "duwei";
    printf("&src out = %p\n",&src);
    printf("src out = %p\n",src);
    strcpy_best2(src,dest);
    printf("dest = %s\n", dest);
    printf("src = %s\n",src);
    return 0;
}

output:
            &src out = 0x7ffeed653a32
            src out = 0x7ffeed653a32
            &src in = 0x7ffeed653a08
            src = 0x7ffeed653a32,line = 64
            dest = dAwei
            src = dAwei

 结论：数组名是常量，不能够被修改。

    • char *strtok(char *str, const char *sep);

      sep参数是字符串，包含分隔符的字符集合。

      #include <stdio.h>
#include <string.h>

int main()
{
    char str[] = "-this,a simple string.";
    char *pch;
    printf("Splitting string \"%s\" into tokens:\n",str);
    pch = strtok(str, ",.-");
    while (pch != NULL){
        printf("%s\n",pch);
        pch = strtok(NULL,",.-");
    }
    return 0;
}

      strtok会改变被操作字符串，所以一般先进行拷贝再操作

    }

}

    • ||、&&优先级

      #include <stdio.h>
int main(){
    int a=1,b=1,c=1;
    if (a-- || b-- && --c){
        printf("%d,%d,%d\n",a,b,c);
    } else{
        printf("%d,%d,%d\n",a,c,b);
    }
}

/*
 * 逻辑与(&&)的优先级比逻辑或(||)的优先级高
 * 先进行 b-- && --c
 * 再将其结果和a--进行逻辑或
 * */

    • switch中使用continue

      #include <stdio.h>
int main()
{
    int a = 123456,b;
    while (a){
        b=a%10;
        a/=10;
        switch (b) {
            default:printf("%d",b++);
            case 1:continue;
            case 2:printf("%d",b++);continue;
            case 3:printf("%d",b++);
            case 4:printf("%d",b++);continue;
        }
    }
    return 0;
}
/*
 * 这里使用continue会导致switch所有分支在执行完成之后返回到外层的while循环，
 * 因此default不会被执行
 * */

    • 局部变量覆盖全局变量

      
#include <stdio.h>
int a=1;
int func(int d)
{
    int b=1;
    static int c =1;
    a++;b++;++c;++d;
    return a+b+c+d;
}

int main()
{
    int k,a=4;
    for (k = 0; k < 3; ++k) {
        printf("%d,",func(a));
    }
    return 0;
}

/*
 * 在main()函数作用域中，局部变量a屏蔽全局变量a。
 * func()函数和main()函数作用域属于同级，func()中没有a变量，所以去全局中找
 * */
